git has 3 stages
- working files
- staging
- commit

-git configure-
git config --global user.name "Adam"
git config --global user.email steve@gmal.com
git config --global init.default branch main
git config -h

-initialize repository-
cd c:/users/steve/desktop/projects
git init

-Git status-
git status
git add filename.txt
git rm --cached filename.text
clear

-Ignore files-
create a textile named .gitignore
within the file add 
# ignore all .txt files
*.txt

-track all files/adding to staging-
git add .

-commit-
git commit -m "my first commit"
git diff
git restore --staged file.txt
